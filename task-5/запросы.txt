CREATE TABLE application5 (
    id INT(10) unsigned NOT NULL AUTO_INCREMENT,
    name VARCHAR(128) NOT NULL DEFAULT '',
    email VARCHAR(30),
    year YEAR(4),
    gender VARCHAR(10),
    kon INT(5),
    bio TEXT,
    login VARCHAR(32),
    pass VARCHAR(64),
    PRIMARY KEY(id)
    );

CREATE TABLE Superpowers5(
    id INT(10) unsigned NOT NULL,
    superpowers INT(10)
    );

    id INT(10) unsigned NOT NULL,
